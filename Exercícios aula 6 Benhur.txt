#include <stdio.h>
#include <math.h>

float heron(float a, float b, float c);

int main()
{
    float L1;
    float L2;
    float L3;
    
    printf("Quais são as medidas dos lados do triângulo (metros)?\n\n");
    
    printf("Lado 1: ");
    scanf("%f", &L1);
    
    printf("Lado 2: ");
    scanf("%f", &L2);
    
    printf("Lado 3: ");
    scanf("%f", &L3);
    
    printf("Área do triângulo: %.2fm²\n", heron(L1, L2, L3));
}

float heron(float a, float b, float c)
{
     float p;
     float R;
     
     p = (a + b + c) / 2;
     
     R = sqrt(p * (p - a) * (p - b) * (p - c));
     
     return R;
}
