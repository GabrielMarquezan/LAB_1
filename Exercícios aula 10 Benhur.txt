#include <stdio.h>

int main()
{
    for(int i = 1; i <= 100; i++)
    {
        printf("%d", i);

        if(i % 2 == 0)
        {
            printf("*");
        }
        
        if(i % 3 == 0)
        {
            printf("@");
        }
        
        if(i % 5 == 0)
        {
            printf("#");
        }
        
        if(i % 7 == 0)
        {
            printf("$");
        }

        printf(" ");
    }

    printf("\n");
}

Exercício 6:

#include <stdio.h>

int quadrado(int numero);
void imprime_char(int numero, char c);

int main()
{
    for(int i = 1; i <= 8; i++)
    {
        imprime_char(quadrado(i), ' ');
        printf("%d\n", i);
    }
}

int quadrado(int numero)
{
    return numero * numero;
}

void imprime_char(int numero, char c)
{
    for(int i = 0; i < numero; i++)
    {
        putchar(c);
    }
}
