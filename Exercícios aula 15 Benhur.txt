Exercício 2:

#include <stdio.h>

int vogais(char a[], char b[]);

int main()
{
    char s1[] = "aei";
    char s2[] = "aeiou";

    printf("%d\n", vogais(s1, s2));
}

int vogais(char a[], char b[])
{
    int cont1 = 0, cont2 = 0;

    for(int i = 0; a[i] != '\0'; i++)
    {
        if(a[i] == 'a' || a[i] == 'A')
        {
            cont1++;
        }
        else if(a[i] == 'e' || a[i] == 'E')
        {
            cont1++;
        }
        else if(a[i] == 'i' || a[i] == 'I')
        {
            cont1++;
        }
        else if(a[i] == 'o' || a[i] == 'O')
        {
            cont1++;
        }
        else if(a[i] == 'u' || a[i] == 'U')
        {
            cont1++;
        }
    }

    for(int i = 0; b[i] != '\0'; i++)
    {
        if(b[i] == 'a' || b[i] == 'A')
        {
            cont2++;
        }
        else if(b[i] == 'e' || b[i] == 'E')
        {
            cont2++;
        }
        else if(b[i] == 'i' || b[i] == 'I')
        {
            cont2++;
        }
        else if(b[i] == 'o' || b[i] == 'O')
        {
            cont2++;
        }
        else if(b[i] == 'u' || b[i] == 'U')
        {
            cont2++;
        }
    }

    if(cont1 == cont2)
    {
        return 0;
    }
    else if(cont1 > cont2)
    {
        return 1;
    }
    else
    {
        return 2;
    }
}

{

            }
        }

}

void troca(char string[])
{
    char aux[2] = ",";
    int pos = 0;

#include <stdio.h>

void remove_char(char string[], int posicao);
int iguais(char a[], char b[], int param);
void remove_decimal(char string[]);

int main()
{
    char string[] = "Se0m 1Se2gm3enta4tio5n F6a7u8l9t";

    printf("%s\n", string);

    remove_decimal(string);

    printf("%s\n", string);
}

void remove_char(char string[], int posicao)
{
    for(int i = posicao; string[i] != '\0'; i++)
    {
        string[i] = string[i + 1];
    }
}

int iguais(char a[], char b[], int param)
{
    for(int i = 0; b[i] != '\0'; i++)
    {
        for(int j = param; a[j] != '\0'; j++)
        {
            if(b[i] == a[j])
            {
                return j;
            }
        }
    }

    return -1;
}

void remove_decimal(char string[])
{
    char decimais[] = "0123456789";
    int pos;

    for(int i = 0; string[i] != '\0'; i++)
    {
        pos = iguais(string, decimais, i);

        remove_char(string, pos);
    }
}
