    {
        printf("Temperatura: %3.2f°F\n", temperatura(t));
        printf("Nossa, você está com febre!\n");
        return 0;
    }
    else if(t >= 37.3)
    {
        printf("Temperatura: %3.2f°F\n", temperatura(t));
        printf("Nossa, você está com febrícula!\n");
        return 0;
    }
    else if(t <= 35)
    {
        printf("Temperatura: %3.2f°F\n", temperatura(t));
        printf("Está muito baixa, é hipotermia!\n");
        return 0;
    }
    else
    {
        printf("Temperatura: %3.2f°F\n", temperatura(t));
        printf("Sua temperatura está normal!\n");
        return 0;
    }
  
}

float temperatura(float c)
{
    float R;
    R = 32 + 1.8 * c;
    
    return R;
}

#include <stdio.h>

void imp_1d(int n);
void imp_2d(int n);
void imp_3d(int n);
void imp_4d(int n);
void imp_5d(int n);


int main()
{
    imp_1d(27945);
    printf("\n");

    imp_2d(27945);
    printf("\n");

    imp_3d(27945);
    printf("\n");

    imp_4d(27945);
    printf("\n");

    imp_5d(27945);
    printf("\n");
}

void imp_1d(int n)
{
    printf("%i", n % 10);
}

void imp_2d(int n)
{
  imp_1d(n / 10);
  imp_1d(n);
}

void imp_3d(int n)
{
  imp_1d(n / 100);
  imp_2d(n);
}

void imp_4d(int n)
{
  imp_1d(n / 1000);
  imp_3d(n);
}

void imp_5d(int n)
{
  imp_1d(n / 10000);
  imp_4d(n);
}
